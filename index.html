Introduction

The Smart IoT Home Automation system is designed to provide convenient and efficient control of household electrical appliances through a web-based interface. By integrating a microcontroller with WiFi communication and relay modules, the system allows users to remotely switch devices ON or OFF in real time. This project demonstrates how IoT technology enhances modern living by improving accessibility, automation, and energy management.

Working Principle

When a user presses a button on the interface, a command is sent to the microcontroller via WiFi. The controller processes the signal and activates the corresponding relay module. The relay switches the connected appliance ON or OFF. Pressing the same button again deactivates the relay. This ensures safe, reliable, and real-time control of home appliances.

Features

• Remote appliance control

• Real-time switching

• Energy efficient

• Web-based interface

• Expandable system

Demo Steps

• Power on the system and connect to WiFi.

• Open the web interface in a browser.

• Use Light 1 and Light 2 to control individually.

• Press All On to turn on all lights.

• Press All Off to turn off all lights.

Source Code



#include <Arduino.h>
#include <ESP8266WiFi.h>
#include <SinricPro.h>
#include <SinricProSwitch.h>

/* ========= USER CONFIG ========= */

#define WIFI_SSID     "TEST_ESP"
#define WIFI_PASS     "12345678"

#define APP_KEY       "47c4524f-eeb0-4d86-947e-66d05b8bf5f8"
#define APP_SECRET    "6c09be76-fde1-4e66-a15f-ea0c410fbf8b-18bb5a2c-2d21-4a36-a099-771b5cfaac2b"

#define DEVICE1_ID    "6973652040cb098d90c4da57"
#define DEVICE2_ID    "697363d57e0883bc2690cd78"
#define DEVICE3_ID    "6973628740cb098d90c4d850"
#define DEVICE4_ID    "xxxxxxxxxxxxxxxxxxxxxxxx"

/* ========= PIN CONFIG ========= */

#define RELAY1 5
#define RELAY2 4
#define RELAY3 14
#define RELAY4 12

#define SW1 13
#define SW2 16
#define SW3 2
#define SW4 15

#define WIFI_LED 16
#define DEBOUNCE_MS 250

struct Device {
  const char* id;
  uint8_t relay;
  uint8_t sw;
  bool lastState;
  unsigned long lastChange;
};

Device devices[] = {
  { DEVICE1_ID, RELAY1, SW1, HIGH, 0 },
  { DEVICE2_ID, RELAY2, SW2, HIGH, 0 },
  { DEVICE3_ID, RELAY3, SW3, HIGH, 0 },
  { DEVICE4_ID, RELAY4, SW4, HIGH, 0 }
};

const uint8_t DEVICE_COUNT = sizeof(devices) / sizeof(devices[0]);

bool onPowerState(const String &deviceId, bool &state) {
  for (uint8_t i = 0; i < DEVICE_COUNT; i++) {
    if (deviceId == devices[i].id) {
      digitalWrite(devices[i].relay, state ? LOW : HIGH);
      return true;
    }
  }
  return false;
}

void setupRelays() {
  for (uint8_t i = 0; i < DEVICE_COUNT; i++) {
    pinMode(devices[i].relay, OUTPUT);
    digitalWrite(devices[i].relay, HIGH);
  }
}

void setupSwitches() {
  for (uint8_t i = 0; i < DEVICE_COUNT; i++) {
    pinMode(devices[i].sw, INPUT_PULLUP);
  }
}

void setupWiFi() {
  WiFi.begin(WIFI_SSID, WIFI_PASS);
  while (WiFi.status() != WL_CONNECTED) {
    delay(300);
  }
  digitalWrite(WIFI_LED, LOW);
}

void setupSinric() {
  for (uint8_t i = 0; i < DEVICE_COUNT; i++) {
    SinricProSwitch &sw = SinricPro[devices[i].id];
    sw.onPowerState(onPowerState);
  }
  SinricPro.begin(APP_KEY, APP_SECRET);
  SinricPro.restoreDeviceStates(true);
}

void handleSwitches() {
  unsigned long now = millis();

  for (uint8_t i = 0; i < DEVICE_COUNT; i++) {
    bool current = digitalRead(devices[i].sw);

    if (current != devices[i].lastState &&
        (now - devices[i].lastChange) > DEBOUNCE_MS) {

      devices[i].lastChange = now;
      devices[i].lastState = current;

      if (current == LOW) {
        bool relayState = !digitalRead(devices[i].relay);
        digitalWrite(devices[i].relay, relayState);

        SinricProSwitch &sw = SinricPro[devices[i].id];
        sw.sendPowerStateEvent(!relayState);
      }
    }
  }
}

void setup() {
  Serial.begin(9600);

  pinMode(WIFI_LED, OUTPUT);
  digitalWrite(WIFI_LED, HIGH);

  setupRelays();
  setupSwitches();
  setupWiFi();
  setupSinric();
}

void loop() {
  SinricPro.handle();
  handleSwitches();
}
